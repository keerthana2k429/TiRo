// ── Claude API: Analyse road image ────────────────────────────────────────
async function analyseRoadImage(base64, mimeType) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: `You are a road quality inspector AI. Analyse the uploaded road image and return ONLY a JSON object (no markdown, no backticks) with these fields:
{
  "severity": "Critical" | "Moderate" | "Minor",
  "issueType": string (e.g. "Pothole / Crater"),
  "confidence": number (0-100),
  "description": string (2-3 sentences describing what you see),
  "urgency": string (one sentence on how urgently it needs repair),
  "safetyRisk": "High" | "Medium" | "Low"
}`,
      messages: [{
        role: "user",
        content: [
          { type: "image", source: { type: "base64", media_type: mimeType, data: base64 } },
          { type: "text", text: "Analyse this road image for damage or issues." }
        ]
      }]
    })
  });
  const data = await res.json();
  const text = data.content?.map(b => b.text || "").join("") || "{}";
  try { return JSON.parse(text.replace(/```json|```/g, "").trim()); }
  catch { return null; }
}

// ── Claude API ─────────────────────────────────────────────────────────────
async function askClaude(messages) {
  const systemPrompt = `You are TiRo AI, a helpful assistant for a road transparency platform.
You help citizens find information about road contractors, budgets, maintenance history, and how to file complaints.

Available roads in the database: ${Object.keys(ROADS_DB).join(", ")}.

Road data: ${JSON.stringify(ROADS_DB, null, 2)}

When users ask about roads:
- Give specific contractor, budget, and engineer details from the data
- If road not in DB, explain you only have data for listed roads
- Help route complaints to the correct authority based on road type
- Be concise, factual, and citizen-friendly
- Use ₹ for Indian roads, £ for UK roads`;

  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: systemPrompt,
      messages,
    }),
  });
  const data = await res.json();
  return data.content?.map((b) => b.text || "").join("") || "I couldn't process that. Please try again.";
}

        padding: "16px 20px",
        position: "sticky",
        top: 0,
        zIndex: 100,
      }}>
        <div style={{ maxWidth: 900, margin: "0 auto", display: "flex", justifyContent: "space-between", alignItems: "center" }}>
          <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
            <div style={{
              width: 36, height: 36,
              background: "linear-gradient(135deg, #f59e0b, #ef4444)",
              borderRadius: 10,
              display: "flex", alignItems: "center", justifyContent: "center",
              fontSize: 18,
            }}>🛣️</div>
            <div>
              <div style={{ fontFamily: "'Syne', sans-serif", fontWeight: 800, fontSize: 20, letterSpacing: -0.5 }}>TiRo</div>
              <div style={{ fontSize: 10, color: "#64748b", letterSpacing: 2, textTransform: "uppercase" }}>Road Transparency Platform</div>
            </div>
          </div>
          <select value={country} onChange={e => setCountry(e.target.value)} style={{
            background: "#1e293b", border: "1px solid #334155", color: "#94a3b8",
            borderRadius: 8, padding: "6px 12px", fontSize: 13, cursor: "pointer"
          }}>
            <option>All</option>
            <option>India</option>
            <option>UK</option>
          </select>
        </div>
      </div>

      {/* Tabs */}
      <div style={{ maxWidth: 900, margin: "0 auto", padding: "16px 20px 0" }}>
        <div style={{ display: "flex", gap: 4, background: "#0f172a", borderRadius: 12, padding: 4, border: "1px solid #1e293b" }}>
          {[
            { id: "chat", label: "🤖 AI Assistant" },
            { id: "roads", label: "🛣️ Road Explorer" },
            { id: "transparency", label: "💰 Budget Tracker" },
          ].map(t => (
            <button key={t.id} onClick={() => setTab(t.id)} style={{
              flex: 1,
              padding: "10px 8px",
              borderRadius: 9,
              border: "none",
              background: tab === t.id ? "linear-gradient(135deg, #f59e0b22, #ef444422)" : "transparent",
              color: tab === t.id ? "#f59e0b" : "#64748b",
              fontWeight: tab === t.id ? 700 : 500,
              fontSize: 13,
              cursor: "pointer",
              borderBottom: tab === t.id ? "2px solid #f59e0b" : "2px solid transparent",
              transition: "all 0.2s",
              fontFamily: "'DM Sans', sans-serif",
            }}>{t.label}</button>
          ))}
        </div>
      </div>

      <div style={{ maxWidth: 900, margin: "0 auto", padding: "20px" }}>

        {/* CHAT TAB */}
        {tab === "chat" && (
          <div style={{ display: "flex", flexDirection: "column", height: "calc(100vh - 220px)" }}>
            <div style={{
              flex: 1,
              overflowY: "auto",
              padding: "8px 0",
              display: "flex",
              flexDirection: "column",
            }}>
              {messages.map(renderMessage)}
              {loading && (
                <div style={{ display: "flex", gap: 6, padding: "12px 16px" }}>
                  {[0, 1, 2].map(i => (

                      width: 8, height: 8, borderRadius: "50%",
                      background: "#f59e0b",
                      animation: `pulse 1.2s ${i * 0.2}s infinite`,
                    }} />
                  ))}
                </div>
              )}
              <div ref={chatEndRef} />
            </div>
            <div style={{ display: "flex", gap: 10, paddingTop: 16, borderTop: "1px solid #1e293b" }}>
              <input
                value={input}
                onChange={e => setInput(e.target.value)}
                onKeyDown={e => e.key === "Enter" && sendMessage()}
                placeholder="Ask about any road, contractor, or budget..."
                style={{ ...inputStyle, flex: 1, borderRadius: 12 }}
              />
              <button onClick={sendMessage} disabled={loading} style={{
                padding: "0 20px",
                background: loading ? "#1e293b" : "linear-gradient(135deg, #f59e0b, #ef4444)",
                border: "none",
                borderRadius: 12,
                color: loading ? "#64748b" : "#000",
                fontWeight: 800,
                cursor: loading ? "not-allowed" : "pointer",
                fontSize: 16,
              }}>➤</button>
            </div>
            <div style={{ display: "flex", gap: 8, marginTop: 10, flexWrap: "wrap" }}>
              {["Who built NH-48?", "Budget for MDR-7?", "File complaint for SH-1", "Status of A2 UK"].map(q => (
                <button key={q} onClick={() => { setInput(q); }} style={{
                  padding: "6px 12px",
                  background: "#0f172a",
                  border: "1px solid #334155",
                  borderRadius: 20,
                  color: "#64748b",
                  fontSize: 12,
                  cursor: "pointer",
                }}>{q}</button>
              ))}
            </div>
          </div>
        )}

        {/* ROADS TAB */}
        {tab === "roads" && (
          <div>
            <input
              value={searchQuery}
              onChange={e => setSearchQuery(e.target.value)}
              placeholder="🔍 Search by road name, contractor, type..."
              style={{ ...inputStyle, width: "100%", fontSize: 15, marginBottom: 16 }}
            />
            <div style={{ display: "grid", gap: 12 }}>
              {filteredRoads.map(road => (
                <div key={road.name} onClick={() => setSelectedRoad(selectedRoad?.name === road.name ? null : road)}
                  style={{
                    background: "linear-gradient(135deg, #0f172a, #1e293b)",
                    border: `1px solid ${selectedRoad?.name === road.name ? "#f59e0b" : "#334155"}`,
                    borderRadius: 14,
                    padding: "16px 20px",
                    cursor: "pointer",
                    transition: "border-color 0.2s",
                  }}>
                  <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
                    <div>
                      <span style={{ fontSize: 11, color: "#f59e0b", fontWeight: 700, letterSpacing: 2 }}>
                        {road.type} · {road.country}
                      </span>
                      <div style={{ fontSize: 16, fontFamily: "'Syne', sans-serif", color: "#f1f5f9", marginTop: 2 }}>{road.name}</div>
                      <div style={{ fontSize: 13, color: "#64748b", marginTop: 2 }}>{road.contractor}</div>
                    </div>
                    <div style={{ display: "flex", flexDirection: "column", alignItems: "flex-end", gap: 8 }}>
                      <StatusBadge status={road.status} />
                      <span style={{ fontSize: 12, color: "#475569" }}>{road.length}</span>
                    </div>
                  </div>
                </div>
              ))}
            </div>
            {selectedRoad && <RoadCard road={selectedRoad} onComplaint={setComplaintRoad} />}
          </div>
        )}

        {/* BUDGET TAB */}
        {tab === "transparency" && (
          <div>
            <h2 style={{ fontFamily: "'Syne', sans-serif", fontSize: 22, marginBottom: 4 }}>Budget Transparency</h2>
            <p style={{ color: "#64748b", fontSize: 14, marginBottom: 20 }}>All figures from official government sources</p>
            {Object.values(ROADS_DB)
              .filter(r => country === "All" || r.country === country)
              .map(road => {
                const pct = Math.round((road.budget.spent / road.budget.sanctioned) * 100);
                const sym = road.budget.currency === "Cr" ? "₹" : "£";
                return (
                  <div key={road.name} style={{
                    background: "linear-gradient(135deg, #0f172a, #1e293b)",
                    border: "1px solid #334155",
                    borderRadius: 14,
                    padding: 20,
                    marginBottom: 14,
                  }}>
                    <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", flexWrap: "wrap", gap: 8 }}>
                      <div>
                        <div style={{ fontSize: 11, color: "#f59e0b", letterSpacing: 2, fontWeight: 700 }}>{road.type} · {road.country}</div>
                        <div style={{ fontSize: 16, color: "#f1f5f9", fontFamily: "'Syne', sans-serif", marginTop: 2 }}>{road.name}</div>
                      </div>
                      <div style={{ textAlign: "right" }}>
                        <div style={{ fontSize: 22, fontWeight: 800, color: pct > 95 ? "#ef4444" : "#22c55e" }}>{pct}%</div>
                        <div style={{ fontSize: 11, color: "#64748b" }}>of budget spent</div>
                      </div>
                    </div>
                    <div style={{ display: "flex", gap: 12, marginTop: 14, flexWrap: "wrap" }}>
                      {[
                        { label: "Sanctioned", val: `${sym}${road.budget.sanctioned} ${road.budget.currency}`, color: "#3b82f6" },
                        { label: "Spent", val: `${sym}${road.budget.spent} ${road.budget.currency}`, color: "#22c55e" },
                        { label: "Remaining", val: `${sym}${road.budget.sanctioned - road.budget.spent} ${road.budget.currency}`, color: "#f59e0b" },
                      ].map(item => (
                        <div key={item.label} style={{
                          flex: 1,
                          minWidth: 100,
                          background: "#060d1a",
                          borderRadius: 10,
                          padding: "10px 14px",
                          border: `1px solid ${item.color}33`,
                        }}>
                          <div style={{ fontSize: 11, color: "#64748b", marginBottom: 4 }}>{item.label}</div>
                          <div style={{ fontSize: 16, fontWeight: 700, color: item.color }}>{item.val}</div>
                        </div>
                      ))}
                    </div>
                    <div style={{ marginTop: 12 }}>
                      <div style={{ background: "#1e293b", borderRadius: 8, height: 8, overflow: "hidden" }}>
                        <div style={{
                          width: `${Math.min(pct, 100)}%`,
                          height: "100%",
                          background: pct > 95 ? "#ef4444" : "linear-gradient(90deg, #3b82f6, #22c55e)",
                          borderRadius: 8,
                        }} />
                      </div>
                    </div>
                    <div style={{ marginTop: 10, fontSize: 11, color: "#475569" }}>
                      📌 {road.budget.source}
                    </div>
                    <button
                      onClick={() => setComplaintRoad(road)}
                      style={{
                        marginTop: 12,
                        padding: "8px 16px",
                        background: "transparent",
                        border: "1px solid #ef444455",
                        borderRadius: 8,
                        color: "#ef4444",
                        fontSize: 13,
                        fontWeight: 600,
                        cursor: "pointer",
                      }}>⚠️ Report Issue</button>
                  </div>
                );
              })}
          </div>
        )}
      </div>

      {complaintRoad && <ComplaintModal road={complaintRoad} onClose={() => setComplaintRoad(null)} />}
    </div>
  );
}
